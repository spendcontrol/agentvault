For Developers

Connect your agent in minutes

Use our Python SDK, integrate with Claude Code, or call contracts directly. Three lines of code to get started.

from yieldvault import YieldVaultAgent

# Initialize agent with your vault
agent = YieldVaultAgent(
    vault_address="0xYourVault...",
    private_key="0xAgentKey...",
    rpc_url="https://sepolia.base.org"
)

# Check available budget
budget = agent.available_yield()
daily_left = agent.remaining_daily_budget()
print(f"Available: {budget} wstETH, Daily left: {daily_left}")

# Spend yield to pay for a service
tx = agent.spend(
    to="0xServiceProvider...",
    amount="0.001"  # wstETH
)

# Swap yield for USDC via Uniswap
tx = agent.swap(
    token_out="0xUSDC...",
    amount_in="0.005",
    min_out="10.0",
    to="0xRecipient..."
)
// Claude Code / LLM Tool Integration
// Give your LLM these tools to manage its own budget:

const tools = [
  {
    name: "check_budget",
    description: "Check available yield balance and daily budget",
    async execute() {
      const vault = new ethers.Contract(VAULT, VAULT_ABI, signer);
      const [principal, , yield_, spent, daily] =
        await vault.getStats();
      return {
        available_yield: ethers.formatEther(yield_),
        daily_remaining: ethers.formatEther(daily),
        total_spent: ethers.formatEther(spent)
      };
    }
  },
  {
    name: "spend_yield",
    description: "Send wstETH yield to an address",
    async execute({ to, amount }) {
      const vault = new ethers.Contract(VAULT, VAULT_ABI, signer);
      const tx = await vault.spend(
        to, ethers.parseEther(amount)
      );
      return await tx.wait();
    }
  }
];
# Check available yield (read-only, no auth needed)
cast call $VAULT \
  "availableYield()(uint256)" \
  --rpc-url https://sepolia.base.org

# Check remaining daily budget
cast call $VAULT \
  "remainingDailyBudget()(uint256)" \
  --rpc-url https://sepolia.base.org

# Spend yield (agent wallet required)
cast send $VAULT \
  "spend(address,uint256)" \
  $RECIPIENT 1000000000000000 \
  --private-key $AGENT_KEY \
  --rpc-url https://sepolia.base.org

# Swap yield for USDC via Uniswap
cast send $VAULT \
  "spendAndSwap(address,uint24,uint256,uint256,address)" \
  $USDC 3000 5000000000000000 0 $RECIPIENT \
  --private-key $AGENT_KEY \
  --rpc-url https://sepolia.base.org

from yieldvault import VaultClient

client = VaultClient(
    rpc_url=os.environ["RPC_URL"],
    vault_address=os.environ["VAULT_ADDRESS"],
    agent_private_key=os.environ["AGENT_PRIVATE_KEY"],
)

# Check how much budget the agent has
budget = client.check_budget()
print(f"Available: {budget['available_yield']} wei")

# Spend yield to pay for something
tx = client.spend(to_address="0xServiceAddress", amount_wei=1000000000000000)

# Or swap yield to USDC first, then pay
tx = client.spend_and_swap(
    token_out="0xUSDC_ADDRESS",
    fee=3000,
    amount_in_wei=1000000000000000,
    amount_out_minimum=0,
    to_address=client.address,
)
